def plot_won_lost_by_year(df):
    """
    Plots a stacked bar chart of 'WON' and 'LOST' cases for each year in the dataset.

    Parameters:
    - df: pandas DataFrame containing the election data.
    """
    # Extract year from 'new_date_filed'
    df['year_filed'] = pd.DatetimeIndex(df['new_date_filed']).year
    
    # Group by year and election result, then count occurrences
    grouped_data = df.groupby(['year_filed', 'election_results_final']).size().unstack(fill_value=0)
    
    # Plotting
    grouped_data[['WON', 'LOST']].plot(kind='bar', stacked=True, figsize=(15, 8), color=['green', 'red'])
    plt.title('Election Results by Year')
    plt.xlabel('Year Filed')
    plt.ylabel('Number of Cases')
    plt.xticks(rotation=45)
    plt.legend(title='Election Result')

    plt.show()

def plot_mean_union_representation_by_year(df):
    """
    Plots the average 'mean_union_representation' for each year in the dataset.

    Parameters:
    - df: pandas DataFrame containing the election data.
    """
    # Ensure 'year_filed' is available; if not, extract year from 'new_date_filed'
    if 'year_filed' not in df.columns:
        df['year_filed'] = pd.DatetimeIndex(df['new_date_filed']).year
    

    # Group by year and calculate the mean of 'mean_union_representation'
    yearly_mean_representation = df.groupby('year_filed')['mean_union_representation'].mean()

    # Plotting
    plt.figure(figsize=(15, 8))
    yearly_mean_representation.plot(kind='line', color='blue', marker='o')
    plt.title('Mean Union Representation by Year')
    plt.xlabel('Year Filed')
    plt.ylabel('Mean Union Representation')
    plt.grid(True)
    plt.xticks(rotation=45)

    plt.show()

def plot_mean_union_representation_by_result(df):
    """
    Creates a graph with two lines, one for the yearly average 'mean_union_representation' for 'WON' cases,
    and another for 'LOST' cases.

    Parameters:
    - df: pandas DataFrame containing the election data.
    """
    # Replace 'inf' values with 'NaN' to avoid issues in calculations
    df['mean_union_representation'].replace([np.inf, -np.inf], np.nan, inplace=True)
    
    # Separate the data based on 'election_results_final' and calculate the yearly average
    won_data = df[df['election_results_final'] == 'WON'].groupby('year_filed')['mean_union_representation'].mean()
    lost_data = df[df['election_results_final'] == 'LOST'].groupby('year_filed')['mean_union_representation'].mean()
    
    # Plotting
    plt.figure(figsize=(15, 8))
    plt.plot(won_data, label='WON', color='green', marker='o')
    plt.plot(lost_data, label='LOST', color='red', marker='x')
    
    plt.title('Yearly Mean Union Representation by Election Result')
    plt.xlabel('Year Filed')
    plt.ylabel('Mean Union Representation')
    plt.legend()
    plt.grid(True)
    plt.xticks(rotation=45)

    plt.show()

def plot_days_filed_to_closed_by_result(df):
    """
    Creates a graph with two lines, one for the yearly average 'Days_Filed_to_Closed' for 'WON' cases,
    and another for 'LOST' cases, differentiating between the outcomes.

    Parameters:
    - df: pandas DataFrame containing the election data.
    """
    # Filter the DataFrame to exclude rows from 1969
    df = df[df['year_filed'] != 1969]
    
    # Calculate the yearly mean of 'Days_Filed_to_Closed' for WON and LOST cases
    won_days_mean = df[df['election_results_final'] == 'WON'].groupby('year_filed')['Days_Filed_to_Closed'].mean()
    lost_days_mean = df[df['election_results_final'] == 'LOST'].groupby('year_filed')['Days_Filed_to_Closed'].mean()
    
    # Plotting
    plt.figure(figsize=(15, 8))
    plt.plot(won_days_mean, label='WON', color='green', marker='o')
    plt.plot(lost_days_mean, label='LOST', color='red', marker='x')
    
    plt.title('Yearly Mean Days from Filed to Closed by Election Result')
    plt.xlabel('Year Filed')
    plt.ylabel('Mean Days Filed to Closed')
    plt.legend()
    plt.grid(True)
    plt.xticks(rotation=45)

    plt.show()
